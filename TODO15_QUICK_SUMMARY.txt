WHAT YOU NEED TO IMPLEMENT:
═══════════════════════════════════════════════════════════════════════════════

1. LABEL DECLARATIONS
   Syntax: label: statement
   Action: Register label in symbol table with SK_LABEL kind
   Assembly: Emit "label:" in code at that position

2. GOTO STATEMENTS  
   Syntax: GOTO label
   Action: Jump to the labeled location
   Assembly: JMP _A60L_label

3. GO TO SYNONYMOUS
   Syntax: GO TO label (two separate keywords)
   Action: Same as GOTO
   Assembly: JMP _A60L_label

4. INDIRECT GOTO
   Syntax: GOTO switch[expr]
   Action: Evaluate expression, use as index into switch array of labels
   Assembly: Load address from switch[index] into BX, then JMP BX

5. SCOPING
   Labels are local to BEGIN/END blocks
   Cannot jump into a block from outside
   Enforced by existing SymEnterScope/SymLeaveScope mechanism

Procedure ParseGotoStmt;
Var
  TargetName : TString;
  SIdx : Integer;
  E : TExprVal;
Begin
  { Case 1: GOTO identifier (label) }
  If CurToken.Kind = TK_IDENT Then
  Begin
    TargetName := CurToken.Val;
    SIdx := SymFind(TargetName);
    
    If SIdx >= 0 Then
    Begin
      If SymTab[SIdx].Kind = SK_LABEL Then
      Begin
        ParserAdvance;  { consommer identifiant }
        EmitComment('GOTO ' + TargetName);
        EmitInstr('JMP', SymTab[SIdx].MangledName);
        Exit;
      End
      Else
      Begin
        CompileError('identifiant ' + TargetName + ' n''est pas une etiquette');
        ParserAdvance;
        Exit;
      End;
    End
    Else
    Begin
      { Might be a forward reference to a label - for now, error }
      CompileError('etiquette ' + TargetName + ' non declaree');
      ParserAdvance;
      Exit;
    End;
  End;

  { Case 2: GOTO expression (indirect - switch array) }
  If (CurToken.Kind = TK_PUNCT) And (CurToken.Val = '(') Then
  Begin
    ParserAdvance;  { consommer ( }
    ParseExpr(E);
    If (CurToken.Kind = TK_PUNCT) And (CurToken.Val = ')') Then
      ParserAdvance
    Else
      CompileError(') attendu');
    
    EmitComment('GOTO indirect (switch expression)');
    ExprFlushToAX(E);
    { TODO: Load switch array base address into BX and emit JMP BX }
    Exit;
  End;

  CompileError('etiquette ou expression attendue apres GOTO');
End;
